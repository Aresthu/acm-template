% 控制列表的间距
% 自定义: \begin{itemize}\setlength{\itemsep}{-\itemsep}
\makeatletter
	\def\enumerate{%
		\ifnum \@enumdepth >\thr@@\@toodeep\else
			\advance\@enumdepth\@ne
			\edef\@enumctr{enum\romannumeral\the\@enumdepth}%
		\expandafter
		\list
			\csname label\@enumctr\endcsname
			{\usecounter\@enumctr\def\makelabel##1{\hss\llap{##1}}%
				%\addtolength{\leftmargin}{5pt}
				\addtolength{\parsep}{10pt}
				%\addtolength{\rightmargin}{20pt} %%%%
				\addtolength{\listparindent}{10pt} %%%%
				\addtolength{\itemsep}{-15pt} %%%%
				\addtolength{\topsep}{-5pt}} %%%%
	\fi}

	\def\itemize{%
		\ifnum \@itemdepth >\thr@@\@toodeep\else
			\advance\@itemdepth\@ne
			\edef\@itemitem{labelitem\romannumeral\the\@itemdepth}%
		\expandafter
		\list
			\csname\@itemitem\endcsname
			{\def\makelabel##1{\hss\llap{##1}}%
				%\addtolength{\leftmargin}{5pt}
				\addtolength{\parsep}{10pt}
				%\addtolength{\rightmargin}{20pt} %%%%
				\addtolength{\listparindent}{10pt} %%%%
				\addtolength{\itemsep}{-12pt} %%%%
				\addtolength{\topsep}{-37pt}} %%%%
	\fi}
\makeatother
